\section{El problema de medir}

% Slide 3
\begin{frame}{¿Cómo mides obesidad?}
  \begin{caso}[Caso clínico]
    Quieres estudiar si la \textbf{obesidad} es un factor de riesgo
    para diabetes mellitus tipo~2 (DM2) en adultos jóvenes de Lima.\\[6pt]
    \textit{¿Qué variable mides?}
  \end{caso}

  \vspace{8pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item Índice de masa corporal (IMC)
        \item Perímetro abdominal
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item Porcentaje de grasa corporal
        \item Presencia de acantosis nigricans
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

% Slide 4
\begin{frame}{El problema}
  Las cuatro medidas aplicadas al \textbf{mismo paciente} dan resultados distintos.

  \vspace{10pt}
  \begin{center}
    \begin{tabular}{lcc}
      \toprule
      \textbf{Medida} & \textbf{Valor} & \textbf{Clasificación} \\
      \midrule
      IMC             & 28.4 kg/m²    & Sobrepeso \\
      Perímetro abd.  & 94 cm         & Riesgo aumentado \\
      \% grasa corp.  & 34\%          & Obesidad \\
      Acantosis       & Presente      & Resistencia a la insulina \\
      \bottomrule
    \end{tabular}
  \end{center}

  \vspace{8pt}
  \begin{center}
    \textcolor{accent}{\textbf{¿Cuál es la «correcta»?}}
  \end{center}
\end{frame}

% Slide 5
\begin{frame}{La respuesta}
  \begin{definicion}[Operacionalizar]
    Definir de forma explícita y reproducible \textbf{cómo} se medirá
    un concepto abstracto en el contexto de un estudio específico.\\[4pt]
    No hay una medida «correcta» universal: la elección depende de
    la pregunta de investigación.
  \end{definicion}

  \vspace{8pt}
  El proceso de elegir y especificar estas medidas se llama
  \textbf{operacionalización}.
\end{frame}

% Slide 6
\begin{frame}{¿Qué es una variable?}
  \begin{definicion}[Variable]
    Atributo observable que \textbf{varía} entre individuos y que
    puede ser medido o clasificado.
  \end{definicion}

  \vspace{8pt}
  \textbf{Ejemplos clínicos:}
  \begin{itemize}
    \item Edad, sexo, nivel de hemoglobina
    \item Diagnóstico de hipertensión (sí/no)
    \item Grado de dolor (ninguno / leve / moderado / severo)
    \item Número de gestaciones previas
  \end{itemize}
\end{frame}

% Slide 7a
\begin{frame}{¿Qué es operacionalizar?}
  \begin{definicion}[Definición operacional]
    Especificación \textbf{exacta} de cómo se medirá un concepto abstracto:
    instrumento, unidad, punto de corte, protocolo.
  \end{definicion}

  \vspace{8pt}
  Una buena definición operacional responde:
  \begin{itemize}
    \item \textbf{¿Qué} instrumento se usa?
    \item \textbf{¿Cómo} se aplica (posición, condiciones, repeticiones)?
    \item \textbf{¿Qué} se registra y en qué unidad?
  \end{itemize}
\end{frame}

% Slide 7b
\begin{frame}{Ejemplo de definición operacional}
  \textbf{Fuerza de prensión con dinamómetro} (Browner et al., 2022, Apéndice~4A):

  \vspace{6pt}
  \begin{nota}
    «Coloque el dinamómetro en la mano derecha con el dial hacia la palma.
    El brazo debe estar flexionado a 90$^{\circ}$ con el antebrazo paralelo al suelo.
    Apriete tan fuerte como pueda mientras cuento hasta tres.
    Registre en kilogramos, al 0.5~kg más cercano.
    Repita el procedimiento para el brazo izquierdo.»
  \end{nota}

  \vspace{6pt}
  \begin{definicion}[¿Por qué es una buena definición?]
    Instrumento específico (dinamómetro) $\cdot$
    posición exacta (90$^{\circ}$, antebrazo paralelo) $\cdot$
    unidad clara (kg, al 0.5) $\cdot$
    protocolo verbal estandarizado y reproducible
  \end{definicion}
\end{frame}

\begin{frame}{Variables categóricas}
  \begin{columns}[T]
    \begin{column}{0.32\textwidth}
      \textbf{\textcolor{accent}{Dicotómica}}\\[4pt]
      Dos categorías\\[4pt]
      \textit{Vivo / muerto}\\
      \textit{Embarazada / no}
    \end{column}
    \begin{column}{0.32\textwidth}
      \textbf{\textcolor{accent}{Nominal}}\\[4pt]
      Categorías sin orden\\[4pt]
      \textit{Tipo de sangre}\\
      \textit{Identidad de género}
    \end{column}
    \begin{column}{0.32\textwidth}
      \textbf{\textcolor{accent}{Ordinal}}\\[4pt]
      Categorías con orden; intervalos no cuantificables\\[4pt]
      \textit{Dolor: ninguno /}\\
      \textit{leve / moderado / severo}
    \end{column}
  \end{columns}

  \vspace{8pt}
  \begin{nota}
    Poder estadístico: nominal/dicotómica $<$ ordinal $<$ discreta $<$ continua
  \end{nota}
\end{frame}

% Slide 11
\begin{frame}{Variables numéricas}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{\textcolor{accent}{Continua}}\\[4pt]
      Infinitos valores posibles en una escala\\[6pt]
      \begin{itemize}
        \item Hemoglobina (g/dL)
        \item Presión arterial (mmHg)
        \item IMC (kg/m²)
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{\textcolor{accent}{Discreta (conteo)}}\\[4pt]
      Número finito de valores, típicamente enteros\\[6pt]
      \begin{itemize}
        \item N.º de gestaciones
        \item N.º de hospitalizaciones
        \item N.º de medicamentos
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{8pt}
  \begin{nota}
    Una variable discreta con $\geq 6$ valores posibles se comporta como continua en análisis.
  \end{nota}
\end{frame}

\section{Precisión y exactitud}

% Slide 13
\begin{frame}{¿Por qué dos médicos obtienen valores distintos?}
  \begin{caso}[Retomamos el caso]
    Dos residentes toman la presión arterial al mismo paciente con el
    mismo tensiómetro, tres veces cada uno. Los valores difieren.\\[4pt]
    ¿Por qué?
  \end{caso}

  \vspace{8pt}
  \textbf{Fuentes de \textcolor{accent}{error aleatorio}:}
  \begin{itemize}
    \item \textbf{Observador:} velocidad de deflación del manguito, técnica
    \item \textbf{Instrumento:} manómetro descalibrado, temperatura
    \item \textbf{Participante:} variación fisiológica, esfuerzo, ansiedad
  \end{itemize}
\end{frame}

% Slide 14
\begin{frame}{Precisión}
  \begin{definicion}[Precisión]
    Grado en que una variable produce el \textbf{mismo valor} cuando se
    mide repetidamente. Sinónimos: reproducibilidad, confiabilidad,
    consistencia.\\[4pt]
    Mayor precisión = menor error aleatorio = \textbf{mayor poder estadístico}.
  \end{definicion}

  \vspace{8pt}
  \textbf{¿Cómo se evalúa?}
  \begin{itemize}
    \item Variables continuas: desviación estándar intra-sujeto, coef.\ de variación
    \item Variables categóricas: porcentaje de acuerdo, \textbf{Kappa de Cohen ($\kappa$)}
          \cite{Cohen1960-kk}, coeficiente de correlación intraclase (CCI)
          \cite{Shrout1979-qv}
  \end{itemize}
\end{frame}

% Slide 15
\begin{frame}{Exactitud}
  \begin{definicion}[Exactitud]
    Grado en que una variable se \textbf{aproxima al valor verdadero}.
    Depende de minimizar el \textbf{sesgo} (error sistemático), además
    de maximizar la precisión.
  \end{definicion}

  \vspace{8pt}
  \textbf{Fuentes de \textcolor{accent}{sesgo}:}
  \begin{itemize}
    \item \textbf{Observador:} tendencia inconsciente a redondear valores
    \item \textbf{Instrumento:} balanza no calibrada $\to$ pesos consistentemente bajos
    \item \textbf{Participante:} sesgo de recuerdo, deseabilidad social
  \end{itemize}

  \vspace{4pt}
  \textbf{¿Cómo se evalúa?} Comparación con un \textbf{estándar de oro}
  (\textit{gold standard}).
\end{frame}

% Slide 15b — Exactitud vs. validez
\begin{frame}[t]{Exactitud vs.\ validez: ¿cuál es la diferencia?}
  \footnotesize
  \begin{tabular}{p{2.8cm} p{4.6cm} p{4.6cm}}
    \toprule
    & \textbf{\textcolor{accent}{Exactitud}} & \textbf{\textcolor{accent}{Validez}} \\
    \midrule
    \textbf{Pregunta} &
      ¿Cuánto me alejo del valor verdadero? &
      ¿Estoy midiendo el fenómeno correcto? \\[4pt]
    \textbf{Tipo de error} &
      Sesgo sistemático &
      Error de constructo \\[4pt]
    \textbf{Se evalúa con} &
      Estándar de oro cuantitativo &
      Tipos de validez \textit{(contenido, constructo\ldots)} \\[4pt]
    \textbf{Ejemplo} &
      Balanza descalibrada (pesos sistemáticamente bajos) &
      Creatinina como indicador de función renal \\
    \bottomrule
  \end{tabular}

  \vspace{6pt}
  \begin{nota}
    Exactitud $\to$ «¿bien medido?» \quad·\quad Validez $\to$ «¿bien elegido?»
  \end{nota}
\end{frame}

\section{Confiabilidad y validez}

% Slide 17
\begin{frame}{Confiabilidad en la práctica}
  \begin{caso}[¿Dos médicos, mismo diagnóstico?]
    Dos internistas examinan de forma independiente a 50 pacientes y
    clasifican cada caso como «anemia» o «no anemia».
    ¿Con qué frecuencia coinciden?
  \end{caso}
  \vspace{-2pt}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Confiabilidad inter-observador}\\[0pt]
      Medida con el \textbf{Kappa de Cohen ($\kappa$)}:\\[0pt]
      \begin{tabular}{ll}
        $\kappa < 0.4$ & Pobre \\
        $0.4$--$0.6$   & Moderada \\
        $0.6$--$0.8$   & Buena \\
        $> 0.8$        & Excelente \\
      \end{tabular}
    \end{column}
    \begin{column}{0.4\textwidth}
      \begin{nota}
        Kappa corrige el acuerdo por azar.\\
        \textit{Ej.:} 80\% observado, 60\% por azar
        $\Rightarrow$ $\kappa = 0.50$ (moderada).\\
        El «\% de acuerdo» es engañoso si una categoría domina.
      \end{nota}
    \end{column}
  \end{columns}
\end{frame}

% Slide 18
\begin{frame}{Validez}
  \begin{definicion}[Validez]
    Dimensión cualitativa de la exactitud: qué tan bien una medición
    \textbf{representa el fenómeno} que pretende medir.
  \end{definicion}

  \vspace{4pt}
  \textbf{Ejemplo (Browner):}

  {\small
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Creatinina sérica}\\
      \textit{Precisa y exacta}\\[2pt]
      Pero: influida por la \textbf{masa muscular}.
      Un paciente con poca masa muscular puede tener
      IRC con creatinina sérica normal.
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Cistatina C}\\
      \textit{Precisa y exacta}\\[2pt]
      \textbf{Más válida} como medida de función renal:
      no depende de masa muscular.
    \end{column}
  \end{columns}
  }
\end{frame}

% Slide 19
\begin{frame}{Tipos de validez}
  \footnotesize
  \begin{tabular}{p{3.2cm} p{6.7cm} p{2.1cm}}
    \toprule
    \textbf{Tipo} & \textbf{Pregunta que responde} & \textbf{Ancla} \\
    \midrule
    \textbf{De contenido} &
      ¿La medida cubre \textit{todos} los aspectos del fenómeno?
      \textit{(SF-36 para calidad de vida)} &
      ¿Cubre todo? \\[3pt]
    \textbf{De apariencia} \textit{(cara)} &
      ¿La medida parece razonable a simple vista?
      \textit{(fragilidad = debilidad + fatiga + pérdida de peso)} &
      ¿Parece correcto? \\[3pt]
    \textbf{De constructo} &
      ¿La medida se comporta como predice la teoría?
      \textit{(aislamiento social separa a quienes tienen pocas amistades)} &
      ¿Predice la teoría? \\[3pt]
    \textbf{De criterio} &
      ¿La medida correlaciona con otra establecida?
      \textit{(nueva prueba de masa muscular vs.\ fuerza de prensión)} &
      ¿Coincide con otro? \\
    \bottomrule
  \end{tabular}
\end{frame}

% Slide 20
\begin{frame}{El estándar de oro (\textit{gold standard})}
  \begin{definicion}[Estándar de oro]
    Medición de referencia que se cree que representa mejor
    el \textbf{valor verdadero} de la característica.
  \end{definicion}

  \vspace{4pt}
  {\small
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Cuando existe:}\\[2pt]
      Permite calcular exactitud cuantitativamente.\\[2pt]
      \textit{Ej.: biopsia para histología de cáncer}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Cuando no existe:}\\[2pt]
      Variables subjetivas (dolor, calidad de vida).\\[2pt]
      Usar tipos de validez de la diapositiva anterior.
    \end{column}
  \end{columns}
  }

  \vspace{4pt}
  {\footnotesize\color{textdark!70}\itshape
    Elegir el \textit{gold standard} puede ser difícil; requiere consenso
    de expertos y revisión de la literatura.}
\end{frame}

\section{Mensajes clave}


% Slide 23a
\begin{frame}{Mensajes clave (1/2)}
  \begin{clave}
    \begin{enumerate}
      \item \textbf{Operacionalizar} es definir exactamente cómo medir
            un concepto abstracto. La definición operacional debe ser
            explícita, reproducible y adecuada a la pregunta de investigación.
      \item Las variables \textbf{continuas} tienen mayor poder estadístico
            que las categóricas: no las dicotomices innecesariamente.
      \item \textbf{Precisión} = sin error aleatorio (reproducibilidad).
            \textbf{Exactitud} = cerca del valor verdadero (sin sesgo).
    \end{enumerate}
  \end{clave}
\end{frame}

% Slide 23b
\begin{frame}{Mensajes clave (2/2)}
  \begin{clave}
    \begin{enumerate}
      \setcounter{enumi}{3}
      \item \textbf{Validez} es qué tan bien la medición representa
            el fenómeno: busca instrumentos ya validados antes de crear uno nuevo.
      \item Toda medición tiene limitaciones. El objetivo es
            \textbf{minimizarlas}, no eliminarlas.
    \end{enumerate}
  \end{clave}
\end{frame}

% Slide 24d — synthesis clave
\begin{frame}{¿Qué nos enseña este caso?}
  \begin{clave}
    \begin{enumerate}
      \item La operacionalización del ingreso \textbf{cambia la pregunta causal},
            no solo la medición: ingreso neto $\neq$ ingreso disponible como
            exposición.
      \item Cada elección en la cadena
            (fuente $\to$ ajuste $\to$ unidad $\to$ comparación)
            afecta precisión, exactitud y validez.
      \item En el \textbf{mapa del estudio} (Slides 2 y 23): lo que decides
            en «Diseño de mediciones» determina directamente la
            inferencia que podrás hacer al final.
    \end{enumerate}
  \end{clave}
