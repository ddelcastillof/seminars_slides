% --- Base theme -----------------------------------------------------------

% --- Color palette --------------------------------------------------------
%  Primary:   Dark navy   #2C3E6B
%  Secondary: Mid blue    #4A6FA5
%  Accent:    Teal        #2E86AB
%  Light:     Light grey  #F4F6F8
%  Text:      Near-black  #1A1A2E

\definecolor{sfPrimary}{HTML}{2C3E6B}
\definecolor{sfSecondary}{HTML}{4A6FA5}
\definecolor{sfAccent}{HTML}{2E86AB}
\definecolor{sfLight}{HTML}{F4F6F8}
\definecolor{sfText}{HTML}{1A1A2E}

% Box accent colors
\definecolor{boxBlue}{HTML}{2C3E6B}
\definecolor{boxRed}{HTML}{C0392B}
\definecolor{boxGreen}{HTML}{1E8449}
\definecolor{boxOrange}{HTML}{D35400}
\definecolor{boxGold}{HTML}{B7950B}

% --- Slide geometry -------------------------------------------------------
% Madrid default is 1em (~4mm) — increase for visual breathing room
\setbeamersize{text margin left=7mm,text margin right=7mm}

% --- Apply colors to Beamer elements --------------------------------------
\setbeamercolor{palette primary}{bg=sfPrimary, fg=white}
\setbeamercolor{palette secondary}{bg=sfSecondary, fg=white}
\setbeamercolor{palette tertiary}{bg=sfAccent, fg=white}
\setbeamercolor{palette quaternary}{bg=sfPrimary, fg=white}
\setbeamercolor{structure}{fg=sfPrimary}
\setbeamercolor{frametitle}{bg=sfPrimary, fg=white}
\setbeamercolor{title}{fg=white}
\setbeamercolor{author}{fg=sfText}
\setbeamercolor{date}{fg=sfText}
\setbeamercolor{institute}{fg=sfText}
\setbeamercolor{block title}{bg=sfSecondary, fg=white}
\setbeamercolor{block body}{bg=sfLight, fg=sfText}
\setbeamercolor{itemize item}{fg=sfSecondary}
\setbeamercolor{itemize subitem}{fg=sfAccent}
\setbeamercolor{enumerate item}{fg=sfSecondary}
\setbeamercolor{section in toc}{fg=sfPrimary}
\setbeamercolor{subsection in toc}{fg=sfSecondary}

% --- Fonts in Beamer elements ---------------------------------------------
\setbeamerfont{title}{size=\LARGE, series=\bfseries}
\setbeamerfont{frametitle}{size=\large, series=\bfseries}
\setbeamerfont{block title}{size=\normalsize, series=\bfseries}

% --- Navigation / footline cleanup ----------------------------------------
\beamertemplatenavigationsymbolsempty
\setbeamertemplate{footline}{%
  \leavevmode%
  \hbox{%
    \begin{beamercolorbox}[wd=.4\paperwidth,ht=2.25ex,dp=1ex,left,leftskip=4pt]{palette primary}%
      \usebeamerfont{author in head/foot}\insertshortauthor
    \end{beamercolorbox}%
    \begin{beamercolorbox}[wd=.4\paperwidth,ht=2.25ex,dp=1ex,center]{palette secondary}%
      \usebeamerfont{title in head/foot}\insertshorttitle
    \end{beamercolorbox}%
    \begin{beamercolorbox}[wd=.2\paperwidth,ht=2.25ex,dp=1ex,right,rightskip=4pt]{palette primary}%
      \usebeamerfont{date in head/foot}%
      \insertframenumber{} / \inserttotalframenumber
    \end{beamercolorbox}%
  }%
  \vskip0pt%
}

% --- Itemize bullet style -------------------------------------------------
\setbeamertemplate{itemize item}{\scriptsize\raise1.25pt\hbox{\donotcoloroutermaths$\blacktriangleright$}}
\setbeamertemplate{itemize subitem}{\scriptsize\raise1.25pt\hbox{\donotcoloroutermaths$\triangleright$}}

% --- keybox: Blue border box — Conceptos clave ----------------------------
\newtcolorbox{keybox}[1][Concepto clave]{
  enhanced,
  colback=sfLight,
  colframe=boxBlue,
  fonttitle=\bfseries\small,
  title=#1,
  boxrule=1.2pt,
  arc=2pt,
  left=4pt, right=4pt, top=4pt, bottom=4pt
}

  colback=white,
  colframe=boxRed,
  fonttitle=\bfseries\small\color{white},

  boxed title style={colback=boxRed, arc=2pt, boxrule=0pt},

  boxrule=0.4pt,

  left=6pt, right=4pt, top=6pt, bottom=4pt
